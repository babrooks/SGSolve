92  void iterate()
93  {
94  try
95  {
96  // Add the extreme tuples array to soln.
97  if (approx.getNumIterations()==0)
98  {
99  for (vector<SGTuple>::const_iterator tuple
100  = approx.getExtremeTuples().begin();
101  tuple != approx.getExtremeTuples().end();
102  ++tuple)
103  soln.push_back(*tuple);
104  }
105  else
106  soln.push_back(approx.getExtremeTuples().back());
107 
108  if (approx.generate() > env.getParam(SG::ERRORTOL)
109  && approx.getNumIterations() < env.getParam(SG::MAXITERATIONS))
110  {
111  status = NOTCONVERGED;
112  emit resultReady(false);
113  }
114  else
115  {
116  approx.end();
117 
118  status = CONVERGED;
119  emit resultReady(true);
120  }
121  }
122  catch (exception & e)
123  {
124  // Add the extreme tuples array to soln.

126 
127  approx.end();
