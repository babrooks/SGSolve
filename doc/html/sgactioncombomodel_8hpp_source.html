74  {
75  if (index.row()>0)
76  {
77  int state = controller->getState();
78  int action = controller->getCurrentIter().getActions()[state][index.row()-1].getAction();
79  const vector< int >& numActions = controller->getSolution()->getGame().getNumActions()[state];
80  QString dataString = QString("A")
81  + QString::number(action)

84  + QString(",C")
85  + QString::number(action/numActions[0])
86  + QString(")");
87  if (state == controller->getCurrentIter().getBestState()
88  && action == controller->getCurrentIter().getBestAction())
89  dataString += QString("*");
90  return dataString;
91  }
92  else if (index.row()==0)
93  return QString("-");
